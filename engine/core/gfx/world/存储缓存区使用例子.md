  1. 简单的数据结构

  // StorageBufferSimple.h
  #pragma once
  #include <vulkan/vulkan.h>
  #include <glm/glm.hpp>
  #include <vector>

  // 物体数据（C++ 端）
  struct ObjectStorageData {
      alignas(16) glm::mat4 modelMatrix;    // 64 bytes
      alignas(16) glm::vec4 color;          // 16 bytes
      alignas(4)  uint32_t materialIndex;   // 4 bytes
      alignas(4)  uint32_t textureIndex;    // 4 bytes
      alignas(4)  uint32_t flags;           // 4 bytes
      alignas(4)  uint32_t padding;         // 4 bytes
      // 总计：96 bytes
  };

  class StorageBufferSimple {
  public:
      StorageBufferSimple(VkDevice device, VkPhysicalDevice getPhysicalDevice);
      ~StorageBufferSimple();

      // 初始化缓冲区
      void init(uint32_t maxObjects);

      // 添加单个物体数据
      void addObject(const ObjectStorageData& object);

      // 批量添加
      void addObjects(const std::vector<ObjectStorageData>& objects);

      // 直接写入指定索引
      void updateObject(uint32_t index, const ObjectStorageData& object);

      // 清空数据（准备下一帧）
      void clear();

      // 获取缓冲区信息
      VkBuffer getBuffer() const { return buffer_; }
      VkDescriptorBufferInfo getDescriptorInfo() const;
      uint32_t getObjectCount() const { return objectCount_; }
      void* getMappedData() const { return mappedData_; }

  private:
      VkDevice device_;
      VkPhysicalDevice physicalDevice_;

      VkBuffer buffer_;
      VkDeviceMemory memory_;
      void* mappedData_;           // 持久映射的指针

      uint32_t maxObjects_;        // 最大物体数量
      uint32_t objectCount_;       // 当前物体数量
      size_t objectSize_;          // 单个物体大小
      size_t alignedSize_;         // 对齐后的大小
      size_t totalBufferSize_;     // 总缓冲区大小

      void createBuffer();
      uint32_t findMemoryType(uint32_t typeFilter, VkMemoryPropertyFlags properties);
  };

  2. 核心实现（重点在这里！）

  // StorageBufferSimple.cpp
  #include "StorageBufferSimple.h"
  #include <cstring>
  #include <stdexcept>
  #include <iostream>

  StorageBufferSimple::StorageBufferSimple(VkDevice device, VkPhysicalDevice getPhysicalDevice)
      : device_(device)
      , physicalDevice_(getPhysicalDevice)
      , buffer_(VK_NULL_HANDLE)
      , memory_(VK_NULL_HANDLE)
      , mappedData_(nullptr)
      , maxObjects_(0)
      , objectCount_(0)
      , objectSize_(sizeof(ObjectStorageData))
      , alignedSize_(sizeof(ObjectStorageData))
      , totalBufferSize_(0)
  {
  }

  void StorageBufferSimple::init(uint32_t maxObjects) {
      maxObjects_ = maxObjects;
      objectSize_ = sizeof(ObjectStorageData);

      // 获取最小对齐要求（通常是 16 或 256）
      VkPhysicalDeviceProperties properties;
      vkGetPhysicalDeviceProperties(physicalDevice_, &properties);
      size_t minAlignment = properties.limits.minStorageBufferOffsetAlignment;

      // 计算对齐后的大小
      alignedSize_ = (objectSize_ + minAlignment - 1) & ~(minAlignment - 1);

      // 总缓冲区大小
      totalBufferSize_ = alignedSize_ * maxObjects_;

      std::cout << "Storage Buffer Config:" << std::endl;
      std::cout << "  Object size: " << objectSize_ << " bytes" << std::endl;
      std::cout << "  Aligned size: " << alignedSize_ << " bytes" << std::endl;
      std::cout << "  Min alignment: " << minAlignment << " bytes" << std::endl;
      std::cout << "  Total buffer: " << (totalBufferSize_ / 1024.0f) << " KB" << std::endl;

      createBuffer();

      // 持久映射（重要：整个程序运行期间保持映射）
      vkMapMemory(device_, memory_, 0, totalBufferSize_, 0, &mappedData_);

      std::cout << "✓ Storage buffer initialized and mapped" << std::endl;
  }

  void StorageBufferSimple::createBuffer() {
      // 创建缓冲区
      VkBufferCreateInfo bufferInfo{};
      bufferInfo.sType = VK_STRUCTURE_TYPE_BUFFER_CREATE_INFO;
      bufferInfo.size = totalBufferSize_;
      bufferInfo.usage = VK_BUFFER_USAGE_STORAGE_BUFFER_BIT;
      bufferInfo.sharingMode = VK_SHARING_MODE_EXCLUSIVE;

      if (vkCreateBuffer(device_, &bufferInfo, nullptr, &buffer_) != VK_SUCCESS) {
          throw std::runtime_error("Failed to create storage buffer!");
      }

      // 获取内存需求
      VkMemoryRequirements memRequirements;
      vkGetBufferMemoryRequirements(device_, buffer_, &memRequirements);

      // 分配内存
      VkMemoryAllocateInfo allocInfo{};
      allocInfo.sType = VK_STRUCTURE_TYPE_MEMORY_ALLOCATE_INFO;
      allocInfo.allocationSize = memRequirements.size;
      allocInfo.memoryTypeIndex = findMemoryType(
          memRequirements.memoryTypeBits,
          VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT | VK_MEMORY_PROPERTY_HOST_COHERENT_BIT
      );

      if (vkAllocateMemory(device_, &allocInfo, nullptr, &memory_) != VK_SUCCESS) {
          throw std::runtime_error("Failed to allocate storage buffer memory!");
      }

      // 绑定缓冲区和内存
      vkBindBufferMemory(device_, buffer_, memory_, 0);
  }

  // ==================== 核心：添加物体数据 ====================

  void StorageBufferSimple::addObject(const ObjectStorageData& object) {
      if (objectCount_ >= maxObjects_) {
          std::cerr << "⚠️ Warning: Storage buffer is full! (max: "
                    << maxObjects_ << ")" << std::endl;
          return;
      }

      // 计算写入位置
      char* mappedPtr = static_cast<char*>(mappedData_);
      size_t offset = objectCount_ * alignedSize_;

      // 直接 memcpy 到映射内存
      memcpy(mappedPtr + offset, &object, objectSize_);

      objectCount_++;

      // 如果使用 HOST_COHERENT，不需要手动 flush
      // 否则需要调用 vkFlushMappedMemoryRanges
  }

  void StorageBufferSimple::addObjects(const std::vector<ObjectStorageData>& objects) {
      for (const auto& obj : objects) {
          addObject(obj);
      }

      std::cout << "📤 Added " << objects.size() << " objects to storage buffer"
                << " (total: " << objectCount_ << ")" << std::endl;
  }

  void StorageBufferSimple::updateObject(uint32_t index, const ObjectStorageData& object) {
      if (index >= maxObjects_) {
          throw std::runtime_error("Object index out of bounds!");
      }

      // 计算写入位置
      char* mappedPtr = static_cast<char*>(mappedData_);
      size_t offset = index * alignedSize_;

      // 直接写入
      memcpy(mappedPtr + offset, &object, objectSize_);

      // 如果这个索引超过当前计数，更新计数
      if (index >= objectCount_) {
          objectCount_ = index + 1;
      }
  }

  void StorageBufferSimple::clear() {
      objectCount_ = 0;
      // 注意：不需要清空内存，下次写入会覆盖
  }

  VkDescriptorBufferInfo StorageBufferSimple::getDescriptorInfo() const {
      VkDescriptorBufferInfo bufferInfo{};
      bufferInfo.buffer = buffer_;
      bufferInfo.offset = 0;
      bufferInfo.range = VK_WHOLE_SIZE;  // 或者 totalBufferSize_
      return bufferInfo;
  }

  uint32_t StorageBufferSimple::findMemoryType(uint32_t typeFilter,
                                               VkMemoryPropertyFlags properties) {
      VkPhysicalDeviceMemoryProperties memProperties;
      vkGetPhysicalDeviceMemoryProperties(physicalDevice_, &memProperties);

      for (uint32_t i = 0; i < memProperties.memoryTypeCount; i++) {
          if ((typeFilter & (1 << i)) &&
              (memProperties.memoryTypes[i].propertyFlags & properties) == properties) {
              return i;
          }
      }

      throw std::runtime_error("Failed to find suitable memory type!");
  }

  StorageBufferSimple::~StorageBufferSimple() {
      if (mappedData_) {
          vkUnmapMemory(device_, memory_);
      }
      if (buffer_ != VK_NULL_HANDLE) {
          vkDestroyBuffer(device_, buffer_, nullptr);
      }
      if (memory_ != VK_NULL_HANDLE) {
          vkFreeMemory(device_, memory_, nullptr);
      }
  }

  // 初始化
  StorageBufferSimple storageBuffer(device, getPhysicalDevice);

  // ==================== 每帧渲染 ====================
  void renderFrame() {
      // 1. 清空上一帧数据
      storageBuffer.clear();

      // 2. 添加所有物体（批量）
      for (size_t i = 0; i < scene.objects.size(); i++) {
          ObjectStorageData data{};
          data.modelMatrix = scene.objects[i].transform;
          data.color = scene.objects[i].color;
          data.materialIndex = scene.objects[i].materialID;
          data.textureIndex = scene.objects[i].textureID;
          data.flags = 0;

          storageBuffer.addObject(data);
      }

      std::cout << "Total objects in buffer: " << storageBuffer.getObjectCount() << std::endl;

      // 3. 绘制
      drawObjects();
  }

  // ==================== 绘制函数 ====================
  void drawObjects() {
      VkCommandBuffer cmd = getCurrentCommandBuffer();

      // 绑定描述符集
      vkCmdBindDescriptorSets(cmd, VK_PIPELINE_BIND_POINT_GRAPHICS,
                             pipelineLayout, 1, 1, &storageDescriptorSet, 0, nullptr);

      // 绘制每个物体
      for (uint32_t i = 0; i < storageBuffer.getObjectCount(); i++) {
          // Push constants：只传索引
          struct PushConstants {
              uint32_t objectIndex;
          } pushConsts = { i };

          vkCmdPushConstants(cmd, pipelineLayout, VK_SHADER_STAGE_VERTEX_BIT,
                            0, sizeof(pushConsts), &pushConsts);

          vkCmdDrawIndexed(cmd, mesh.indexCount, 1, 0, 0, 0);
      }
  }

  关键点总结

  | 操作   | 代码                                          | 说明             |
  |------|---------------------------------------------|----------------|
  | 初始化  | storageBuffer.init(10000)                   | 创建并映射缓冲区       |
  | 添加数据 | storageBuffer.addObject(obj)                | memcpy 到映射内存   |
  | 清空   | storageBuffer.clear()                       | 重置计数器          |
  | 直接访问 | void* ptr = storageBuffer.getMappedData()   | 获取映射指针         |
  | 对齐   | alignedSize = (size + align-1) & ~(align-1) | 处理对齐要求         |
  | 持久映射 | vkMapMemory(..., &mappedData_)              | 避免每帧 map/unmap |
